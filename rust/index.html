fn add(x: int, y: int) -> int {
    // implicit return when no semicolon
    x + y
}

fn main() {
    // Immutable by default
    let x: int = 1;

    // Optional mutability
    let mut y: int = 2;

    // Type inference
    let z = 1i;

    // Literal strings
    let string = "Hello"
}

// Tuple structs

enum Ordering {
    Less,
    Equal,
    Greater,
}

fn cmp(a: int, b: int) -> Ordering {
    if a < b { Less }
    else if a > b { Greater }
    else { Equal }
}

match cmp(x, y) {
    Less    => println!("less"),
    Greater => println!("greater"),
    Equal   => println!("equal"),
}
// match must be exhaustive
